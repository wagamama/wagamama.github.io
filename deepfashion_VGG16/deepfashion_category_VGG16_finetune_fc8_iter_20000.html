def draw_network(prototxt_file):
    net = caffe.proto.caffe_pb2.NetParameter()
    with open(prototxt_file, 'rt') as f:
        text_format.Merge(f.read(), net)
    net_image = display.Image(data=caffe.draw.draw_net(net, 'BT'))
    display.display(net_image)

s = caffe.proto.caffe_pb2.SolverParameter()

s.random_seed = 0xCAFFE
s.type = 'SGD'

s.base_lr = 0.001
s.momentum = 0.9
s.weight_decay = 5e-4

s.net = train_val_prototxt_file
# s.max_iter = 45000

s.test_iter.append(2500)
s.test_interval = 13000

s.snapshot_prefix = train_val_prototxt_file.split('.')[0]
s.snapshot = 13000
s.display = 1000

s.lr_policy = 'step'
s.gamma = 0.1
s.stepsize = 30000

with open(solver_prototxt_file, 'wt') as f:
    f.write(str(s))

max_iter = 20000
test_interval = 100
test_iter = 2500

train_loss = np.zeros(max_iter)
test_acc = np.zeros(int(np.ceil(max_iter / test_interval)))

caffe.set_mode_gpu()
caffe.set_device(0)
solver = caffe.get_solver(solver_prototxt_file)
solver.net.copy_from(pretrained_VGG16_caffemodel_file)

for it in range(max_iter):
    solver.step(1)
    
    train_loss[it] = solver.net.blobs['loss'].data
    
    if it % test_interval == 0:
        accuracy = 0
        for test_it in range(test_iter):
            solver.test_nets[0].forward()
            accuracy += solver.test_nets[0].blobs['accuracy'].data
        test_acc[int(it // test_interval)] = accuracy / test_iter
#         print('Iteration {} testing...'.format(it))
#         print('   train loss: {:.3f}'.format(train_loss[it]))
#         print('   test acc: {:.3f}'.format(test_acc[int(it // test_interval)]))
        
del solver

_, ax1 = plt.subplots()
ax2 = ax1.twinx()
ax1.plot(np.arange(max_iter), train_loss)
ax2.plot(test_interval * np.arange(len(test_acc)), test_acc, 'r')
ax1.set_xlabel('iteration')
ax1.set_ylabel('train loss')
ax2.set_ylabel('test accuracy')
ax2.set_title('Custom Test Accuracy: {:.2f}'.format(test_acc[-1]))
fig = matplotlib.pyplot.gcf()
fig.set_size_inches(20, 10)
