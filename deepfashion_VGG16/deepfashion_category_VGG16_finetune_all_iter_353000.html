def draw_network(prototxt_file):
    net = caffe.proto.caffe_pb2.NetParameter()
    with open(prototxt_file, 'rt') as f:
        text_format.Merge(f.read(), net)
    net_image = display.Image(data=caffe.draw.draw_net(net, 'BT'))
    display.display(net_image)

pretrained_VGG16_caffemodel_file = 'VGG16/train_val_all_iter_203000.caffemodel'

max_iter = 150000
test_interval = 13000
test_iter = 2500

s = caffe.proto.caffe_pb2.SolverParameter()

s.random_seed = 0xCAFFE
s.type = 'SGD'

s.base_lr = 1e-3
s.momentum = 0.9
s.weight_decay = 5e-4

s.net = train_val_prototxt_file
s.max_iter = max_iter

s.test_iter.append(test_iter)
s.test_interval = test_interval

s.snapshot_prefix = pretrained_VGG16_caffemodel_file.split('.')[0]
s.snapshot = 13000
s.display = 1000

s.lr_policy = 'step'
s.gamma = 0.1
s.stepsize = 30000

with open(solver_prototxt_file, 'wt') as f:
    f.write(str(s))

train_loss = np.zeros(max_iter)
test_acc = np.zeros(int(np.ceil(max_iter / test_interval)))

caffe.set_mode_gpu()
caffe.set_device(0)
solver = caffe.get_solver(solver_prototxt_file)
solver.net.copy_from(pretrained_VGG16_caffemodel_file)

for it in range(max_iter):
    solver.step(1)
    
    train_loss[it] = solver.net.blobs['loss'].data
    
    if it % test_interval == 0:
        accuracy = 0
        for test_it in range(test_iter):
            solver.test_nets[0].forward()
            accuracy += solver.test_nets[0].blobs['accuracy'].data
        test_acc[int(it // test_interval)] = accuracy / test_iter
#         print('Iteration {} testing...'.format(it))
#         print('   train loss: {:.3f}'.format(train_loss[it]))
#         print('   test acc: {:.3f}'.format(test_acc[int(it // test_interval)]))

solver.net.save('{}_iter_{}.caffemodel'.format(pretrained_VGG16_caffemodel_file.split('.')[0], max_iter))
del solver

_, ax1 = plt.subplots()
ax2 = ax1.twinx()
ax1.plot(np.arange(max_iter), train_loss)
ax2.plot(test_interval * np.arange(len(test_acc)), test_acc, 'r')
ax1.set_xlabel('iteration')
ax1.set_ylabel('train loss')
ax2.set_ylabel('test accuracy')
ax2.set_title('Train loss: {:.2f} / Custom Test Accuracy: {:.2f}'.format(train_loss[-1], test_acc[-1]))
fig = matplotlib.pyplot.gcf()
fig.set_size_inches(20, 10)

topk = [1, 3, 5]
ldb_test_data_category = './leveldb/deepfashion_test_data_category_ldb'
finetuned_VGG16_caffemodel_file = 'VGG16/train_val_all_iter_353000.caffemodel'

acc = np.zeros((len(topk),))

caffe.set_mode_gpu()
caffe.set_device(0)
net = caffe.Net(deploy_prototxt_file, finetuned_VGG16_caffemodel_file, caffe.TEST)
mu = np.load('VGG16/mean.npy')
mu = mu.mean(1).mean(1)

net.blobs['data'].reshape(1, 3, net.blobs['data'].data.shape[2], net.blobs['data'].data.shape[3])

transformer = caffe.io.Transformer({'data': net.blobs['data'].data.shape})
transformer.set_transpose('data', (2,0,1))
transformer.set_mean('data', mu)

db = leveldb.LevelDB(ldb_test_data_category)
datum = caffe.proto.caffe_pb2.Datum()
count = 0
for datum_str in db.RangeIter():
    datum.ParseFromString(datum_str[1])
    data = caffe.io.datum_to_array(datum)
    data = np.transpose(data, (1, 2, 0))
    data = cv.resize(data, (net.blobs['data'].data.shape[2],net.blobs['data'].data.shape[3]), interpolation=cv.INTER_CUBIC)
    net.blobs['data'].data[...] = np.asarray([transformer.preprocess('data', data)])

    net.forward()
    output = net.blobs['softmax'].data[0]
    output_ids = output.argsort()[::-1][:topk[-1]]

    for i,k in enumerate(topk):
        if datum.label in output_ids[:k]:
            acc[i] += 1
    count += 1

del db
del net

acc = acc / count
    
for i,k in enumerate(topk):
    print('Top-{} accuracy: {}'.format(k, acc[i]))
